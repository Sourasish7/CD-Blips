![Cd-blips](https://github.com/Sourasish7/CD-Blips/assets/135587815/3d4945ac-51bb-427d-a905-5485d252b3c2)

```lua
 [1] = {
    vector = vector3(0, 0, 0),
    text = "Blip Name", 
    color = 3, 
   sprite = Blip ID, 
    scale = size,
 },
 ```
